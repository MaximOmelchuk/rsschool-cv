Code example

Task: Range Extraction

Solution: 



function solution(list){
    list=list.map((x,i)=>{
        if (x+1!==list[i+1]) {
            return x;
        } else {
            let last=list.slice(i)
                         .findIndex((y,j)=>y+1!==list[j+i+1]);
        if (last>=2) {
            let tmp=x+'-'+list[last+i];
            for (let e=i+1; e<=last+i; e++) {
                list[e]='';
            }
            return tmp;
        } else return x;
        }
    })
    return list.join(' ').trim().split(/ +/).join();
}
